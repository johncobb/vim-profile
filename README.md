
# Navigate to home profile
cd ~/

# Install vim-profile
git clone https://github.com/johncobb/vim-profile.git

# Copy .vimrc to home folder
cp .vimrc ~/

# Install vundle
git clone https://github.com/VundleVim/Vundle.vim.git ~/.vim/bundle/Vundle.vim
# Launch VIM
vim

# Install the Plugins
:PluginInstall

# GtG!
